RE_Norse_spread_norse_faith = {
	change_religion = norse_pagan_reformed
	clr_province_flag = RE_Norse_current_infected
	if = {
		limit = {
			any_neighbor_province = {
				owned_by = ROOT
				NOT = {
					religion = norse_pagan_reformed
				}
			}
		}
		every_neighbor_province = {
			limit = {
				owned_by = ROOT
				NOT = {
					religion = norse_pagan_reformed
				}
			}
			set_province_flag = RE_Norse_current_infected
		}
	}
	else = {
		owner = {
			random_owned_province = {
				limit = {
					NOT = {
						religion = norse_pagan_reformed
					}
				}
				set_province_flag = RE_Norse_current_infected
			}
		}
	}
	add_province_modifier = {
		name = religious_zeal_at_conv
		duration = 3650
	}
}

RE_Norse_determine_gender_loc_for_ruler = {
	if = {
		limit = {
			is_female = yes
		}
		set_country_flag = RE_Norse_previous_she
		set_country_flag = RE_Norse_previous_her
	}
	else = {
		set_country_flag = RE_Norse_previous_he
		set_country_flag = RE_Norse_previous_him
		set_country_flag = RE_Norse_previous_his
	}
}
